376  inline std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw,
377  std::error_code& ec) {
378  ec.clear();
379 #if OPENSSL_VERSION_NUMBER <= 0x10100003L
380  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
381  BIO_new_mem_buf(const_cast<char*>(certstr.data()), static_cast<int>(certstr.size())), BIO_free_all);
382 #else
383  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
384  BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
385 #endif
386  std::unique_ptr<BIO, decltype(&BIO_free_all)> keybio(BIO_new(BIO_s_mem()), BIO_free_all);
387  if (!certbio || !keybio) {
388  ec = error::rsa_error::create_mem_bio_failed;
389  return {};
390  }
391 
392  std::unique_ptr<X509, decltype(&X509_free)> cert(
393  PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);

396  return {};
397  }

399  if (!key) {
400  ec = error::rsa_error::get_key_failed;
401  return {};
402  }

404  ec = error::rsa_error::write_key_failed;

409  if (len <= 0 || ptr == nullptr) {

411  return {};
412  }
